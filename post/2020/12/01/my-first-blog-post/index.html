Finding k-mers, or all the possible, unique sub-sequences from two strings of nucleotide bases.

my_seq1 = "ATCATCATG"
my_seq2 = "CAGCCCAATCAGGCTCTACTGCCACTAAACTTACGCAGGATATATTTACGCCGACGTACT"

def kmers(dna):  #argument "sentence" will be string object
    counts = {}  #new dict
    for i in range(0,len(dna)-2):
        if dna[i:i+3] in counts:     #have we seen it before in this sentence?
            counts[dna[i:i+3]]+= 1   #if yes, increase count by 1
        else:
            counts[dna[i:i+3]] = 1   #otherwise, set count
    return(counts)
        
kmers(my_seq1)
## {'ATC': 2, 'TCA': 2, 'CAT': 2, 'ATG': 1}
